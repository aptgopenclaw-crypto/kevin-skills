#!/bin/bash
# 快速啟動指南 - Windows PowerShell 或 CMD

# 本文件提供快速啟動項目的命令

# ============================================================
# 1. 首次設置（僅需執行一次）
# ============================================================

# 建立虛擬環境
python -m venv venv

# 激活虛擬環境（Windows）
venv\Scripts\activate

# 激活虛擬環境（macOS/Linux）
# source venv/bin/activate

# 安裝所有依賴
pip install -r requirements.txt

# ============================================================
# 2. 運行爬蟲（每次使用）
# ============================================================

# 方式 1：使用交互式選擇菜單（推薦）
python main.py

# 方式 2：直接運行 Requests 版爬蟲
python src/scraper.py

# 方式 3：運行 Selenium 版爬蟲（需要額外安裝）
# pip install selenium webdriver-manager
# python src/scraper_selenium.py

# ============================================================
# 3. 環境測試
# ============================================================

# 檢查環境配置是否正確
python test_env.py

# ============================================================
# 4. 停止虛擬環境
# ============================================================

# 反激活虛擬環境
deactivate

# ============================================================
# 常用命令
# ============================================================

# 升級所有依賴
pip install --upgrade -r requirements.txt

# 列出已安裝的包
pip list

# 凍結當前環境的所有包版本到 requirements_frozen.txt
pip freeze > requirements_frozen.txt

# ============================================================
# 常見問題解決
# ============================================================

# 如果 openpyxl 導入失敗
pip install openpyxl --upgrade

# 如果找不到 Selenium 相關包
pip install selenium webdriver-manager

# 如果 BeautifulSoup 導入失敗
pip install beautifulsoup4 lxml --upgrade

# ============================================================
# 配置修改
# ============================================================

# 編輯 src/config.py 以自訂：
# - KEYWORDS：修改搜索關鍵字
# - BASE_URL：修改政府採購網 URL
# - REQUEST_TIMEOUT：修改請求超時時間
# - OUTPUT_FILE：修改輸出檔案名稱

# ============================================================
# 查看結果
# ============================================================

# 爬蟲完成後，Excel 檔案將保存在：
# output/招標資料.xlsx

# 在 Excel 中打開結果
# Windows：start output/招標資料.xlsx
# macOS：open output/招標資料.xlsx
# Linux：libreoffice output/招標資料.xlsx
